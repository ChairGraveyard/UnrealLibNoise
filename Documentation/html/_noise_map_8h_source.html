55  FColor GetColor(float Value)
56  {
57  if (GradientPoints.Num() >= 2)
58  {
59  int indexPos;
60  for (indexPos = 0; indexPos < GradientPoints.Num(); indexPos++)
61  {
62  if (Value < GradientPoints[indexPos].Position) break;
63  }
64 
65  int index0 = FMath::Clamp(indexPos - 1, 0, GradientPoints.Num() - 1);
66  int index1 = FMath::Clamp(indexPos, 0, GradientPoints.Num() - 1);
67 
68  if (index0 == index1) return GradientPoints[index1].Color;
69 
70  float input0 = GradientPoints[index0].Position;
71  float input1 = GradientPoints[index1].Position;
72  float alpha = (Value - input0) / (input1 - input0);
73 
74  FLinearColor Color0 = FLinearColor(GradientPoints[index0].Color);
75  FLinearColor Color1 = FLinearColor(GradientPoints[index1].Color);
76 
77  FLinearColor ReturnColor = FLinearColor::LerpUsingHSV(Color0, Color1, alpha);
78  return ReturnColor.ToFColor(true);
79  }
80 
81  return FColor::Black;
82  }
